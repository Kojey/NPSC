154  if (USART1->SR & USART_FLAG_IDLE) { /* We want IDLE flag only */

160  tmp = USART1->DR; /* bluetooth_read data register */

170 void DMA2_Stream5_IRQHandler(void) {
171  size_t len, tocopy;
172  uint8_t* ptr;
173 
174  /* Check transfer complete flag */
175  if (DMA2->HISR & DMA_FLAG_TCIF5) {
176  DMA2->HIFCR = DMA_FLAG_TCIF5; /* Clear transfer complete flag */
177 
178  /* Calculate number of bytes actually transfered by DMA so far */
184  len = DMA_RX_BUFFER_SIZE - DMA2_Stream5->NDTR;
185  tocopy = INSTRUCTION_SIZE - bluetooth_write; /* Get number of bytes we can copy to the end of buffer */
186 
187  /* Check how many bytes to copy */
188  if (tocopy > len) {
189  tocopy = len;
190  }
191 
192  /* bluetooth_write received data for UART main buffer for manipulation later */
193  ptr = DMA_RX_Buffer;
194  memcpy(&INSTRUCTION_BUFFER[bluetooth_write], ptr, tocopy); /* Copy first part */
195 
196  /* Correct values for remaining data */
197  bluetooth_write += tocopy;
198  len -= tocopy;
199  ptr += tocopy;
200 
201  /* If still data to write for beginning of buffer */
202  if (len) {
203  memcpy(&INSTRUCTION_BUFFER[0], ptr, len); /* Don't care if we override bluetooth_read pointer now */
204  bluetooth_write = len;
205  }
206 
207  /* Prepare DMA for next transfer */
208  /* Important! DMA stream won't start if all flags are not cleared first */
209  DMA2->HIFCR = DMA_FLAG_DMEIF5 | DMA_FLAG_FEIF5 | DMA_FLAG_HTIF5 | DMA_FLAG_TCIF5 | DMA_FLAG_TEIF5;
210  DMA2_Stream5->M0AR = (uint32_t)DMA_RX_Buffer; /* Set memory address for DMA again */
211  DMA2_Stream5->NDTR = DMA_RX_BUFFER_SIZE; /* Set number of bytes to receive */
212  DMA2_Stream5->CR |= DMA_SxCR_EN; /* Start DMA transfer */
213  }
214 }
