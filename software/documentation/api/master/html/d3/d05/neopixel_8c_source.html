245 void neopixel_setPixelColourRGB(uint8_t n, uint8_t r, uint8_t g, uint8_t b){
246  // scale according to brightness
247  float _max = (float)max(r,g,b);
248  float b_scale = brightness/_max;
249  // scale RGB
250  r = (uint8_t) ((float)r*b_scale);
251  g = (uint8_t) ((float)g*b_scale);
252  b = (uint8_t) ((float)b*b_scale);
253 
254  uint8_t tempBuffer[24];
255  uint32_t i;
256  uint32_t LEDindex;
257  LEDindex = n % LED_NUMBER;
258 
259  for (i = 0; i < 8; i++) // GREEN data
260  tempBuffer[i] = ((g << i) & 0x80) ? WS2812_1 : WS2812_0;
261  for (i = 0; i < 8; i++) // RED
262  tempBuffer[8 + i] = ((r << i) & 0x80) ? WS2812_1 : WS2812_0;
263  for (i = 0; i < 8; i++) // BLUE
264  tempBuffer[16 + i] = ((b << i) & 0x80) ? WS2812_1 : WS2812_0;
265  // stop timer before updating the buffer
266  TIM_Cmd(TIM2,DISABLE);
267  for (i = 0; i < 24; i++)
268  LEDbuffer[RESET_SLOTS_BEGIN + LEDindex * 24 + i] = tempBuffer[i];
269  TIM_Cmd(TIM2,ENABLE);
270 }
