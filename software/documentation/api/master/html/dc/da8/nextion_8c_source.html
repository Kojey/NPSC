120 void USART2_IRQHandler(void) {
121  /* Check for IDLE flag */
122  if (USART2->SR & USART_FLAG_IDLE) { /* We want IDLE flag only */
123  /* This part is important */
124  /* Clear IDLE flag by reading status register first */
125  /* And follow by reading data register */
126  volatile uint32_t tmp; /* Must be volatile to prevent optimizations */
127  tmp = USART2->SR; /* nextion_read status register */
128  tmp = USART2->DR; /* nextion_read data register */
129  (void)tmp; /* Prevent compiler warnings */
130  DMA1_Stream5->CR &= ~DMA_SxCR_EN; /* Disabling DMA will force transfer complete interrupt if enabled */
131  }

137 void DMA1_Stream5_IRQHandler(void) {
138  if (DMA1->HISR & DMA_FLAG_TCIF5) {
139  DMA1->HIFCR = DMA_FLAG_TCIF5; /* Clear transfer complete flag */
140 
141  /* Extract all instruction from DMA */
142  for(uint8_t index=0; index<DMA_MAX_INSTRUCTION; ++index){
143  /* Check for validity of instruction */
144  if(instruction_valid(DMA_RX_Buffer[index*INSTRUCTION_SIZE])
145  && !instruction_ack(&DMA_RX_Buffer[index*INSTRUCTION_SIZE])){
146  InstructionTypeDef newInstruction;
147  for(int i=0; i<INSTRUCTION_SIZE; i++){
148  newInstruction.instrution[i]=DMA_RX_Buffer[index*INSTRUCTION_SIZE+i];
149  // clear DMA_RX_Buffer
150  DMA_RX_Buffer[index*INSTRUCTION_SIZE+i]=0;
151  }
152  newInstruction.excecuted=false;
153  // add it to the queue
154  InstructionQueue_enqueue(instruction_queue,&newInstruction);
155  }
156  }
157 
158  /*Prepare DMA for next transfer
159  Important! DMA stream won't start if all flags are not cleared first */
160  DMA1->HIFCR = DMA_FLAG_DMEIF5 | DMA_FLAG_FEIF5 | DMA_FLAG_HTIF5 | DMA_FLAG_TCIF5 | DMA_FLAG_TEIF5;
161  DMA1_Stream5->M0AR = (uint32_t)DMA_RX_Buffer; /* Set memory address for DMA again */
162  DMA1_Stream5->NDTR = DMA_RX_BUFFER_SIZE; /* Set number of bytes to receive */
163  DMA1_Stream5->CR |= DMA_SxCR_EN; /* Start DMA transfer */
164  }
