45 void instruction_execute(void){
46  RTC_ClockTypeDef clock;
47  AlarmTypeDef alarm;
48  // check if the queue is empty
49  while(instruction_queue->size){
50  // fecth instruction
51  InstructionTypeDef _instruction = InstructionQueue_dequeue(instruction_queue);
52  // decode and execute instruction
53  switch(_instruction.instrution[0]){
54  /* External RTC instructions */
55  case 0x01:
56  // set clock
57  clock.date.RTC_Year=_instruction.instrution[1];
58  clock.date.RTC_Month=_instruction.instrution[2];
59  clock.date.RTC_Date=_instruction.instrution[3];
60  clock.date.RTC_WeekDay=_instruction.instrution[4];
61  clock.time.RTC_Hours=_instruction.instrution[5];
62  clock.time.RTC_Minutes=_instruction.instrution[6];
63  clock.time.RTC_Seconds=_instruction.instrution[7];
64  clock.time.RTC_H12=RTC_H12_AM;
65  rtc_setClockStruct(&clock);
66  break;
67  case 0x02:
68  // get clock
69  clock = rtc_getClockStruct();
70  if(b){
71  alarm_synchronize();
72  b=false;
73  }
74  instruction_nextionStart();
75  instruction_nextionSendInt("home.n0.val=",clock.time.RTC_Hours);
76  instruction_nextionSendInt("home.n1.val=",clock.time.RTC_Minutes);
77  instruction_nextionSendStr("home.t1.txt=", rtc_dayToString(clock.date.RTC_WeekDay) );
78  instruction_nextionSendInt("home.n3.val=",clock.date.RTC_Date);
79  instruction_nextionSendStr("home.t2.txt=", rtc_monthToString(clock.date.RTC_Month) );
80  instruction_nextionSendInt("home.n4.val=",2000+clock.date.RTC_Year);
81  instruction_nextionStop();
82  neopixel_setAllPixelRGB(0,255,0);
83  break;
84  /* Alarm instructions */
85  case 0x10:
86  alarm = alarm_load(_instruction.instrution[1]);
87  // set alarm time params
88  //alarm.id=_instruction.instrution[1];
89  alarm.alarm.RTC_AlarmTime.RTC_Hours=_instruction.instrution[2];
90  alarm.alarm.RTC_AlarmTime.RTC_Minutes=_instruction.instrution[3];
91  alarm.alarm.RTC_AlarmDateWeekDaySel=rtc_weekdaySelTo32Bits(_instruction.instrution[4]);
92  alarm.alarm.RTC_AlarmDateWeekDay=_instruction.instrution[5];
93  alarm.repeat=_instruction.instrution[6];
94  alarm.alarm.RTC_AlarmMask=rtc_alarmMaskTo32Bits(_instruction.instrution[6]);
95  alarm.alarm.RTC_AlarmTime.RTC_H12=RTC_H12_AM;
96  alarm.alarm.RTC_AlarmTime.RTC_Seconds=0;
97  break;
98  case 0x11:
99  // set alarm extra params
100  alarm.ringtone=_instruction.instrution[2];
101  alarm.pattern=_instruction.instrution[3];
102  alarm.enable=_instruction.instrution[4];
103  alarm.fetched=_instruction.instrution[5];
104  break;
105  case 0x12:
106  // set alarm label
107  // get string from instruction
108  get_stringFromInstruction((char *)label_instruction, (char *)&_instruction.instrution[4],LABEL_INSTR_SIZE);
109  if (_instruction.instrution[2]==0)
110  strcpy(label,label_instruction);
111  else {
112  strcat(label,label_instruction);
113  strcpy(alarm.label,label);
114  // last instruction
115  if(_instruction.instrution[2]==2){
116  // save alarm
117  alarm_save(&alarm);
118  alarm_update(true);
119  }
120  }
121  break;
122  case 0x13:
123  // set alarm enable
124  alarm = alarm_load(_instruction.instrution[1]);
125  alarm.enable = _instruction.instrution[2];
126  alarm_save(&alarm);
127  break;
128  case 0x14:
129  // get alarm min params
130  instruction_nextionStart();
131  // update first alarm
132  alarm = alarm_load(_instruction.instrution[1]);
133  instruction_nextionSendInt("n0.val=",alarm.alarm.RTC_AlarmTime.RTC_Hours);
134  instruction_nextionSendInt("n1.val=",alarm.alarm.RTC_AlarmTime.RTC_Minutes);
135  instruction_nextionSendStr("t1.txt=",alarm.label);
136  instruction_nextionSendInt("bt0.val=",alarm.enable);
137  // update second alarm
138  alarm = alarm_load(_instruction.instrution[2]);
139  instruction_nextionSendInt("n2.val=",alarm.alarm.RTC_AlarmTime.RTC_Hours);
140  instruction_nextionSendInt("n3.val=",alarm.alarm.RTC_AlarmTime.RTC_Minutes);
141  instruction_nextionSendStr("t2.txt=",alarm.label);
142  instruction_nextionSendInt("bt1.val=",alarm.enable);
143  instruction_nextionStop();
144  neopixel_setAllPixelRGB(255,0,0);
145  break;
146  case 0x15:
147  // get alarm params
148  instruction_nextionStart();
149  alarm = alarm_load(_instruction.instrution[1]);
150  instruction_nextionSendStr("t6.txt=",alarm_repeatToString(alarm.repeat));
151  instruction_nextionSendStr("t7.txt=",alarm_ringToneToString(alarm.ringtone));
152  instruction_nextionSendStr("t8.txt=",alarm_patternToString(alarm.pattern));
153  instruction_nextionSendStr("t9.txt=",alarm.label);
154  instruction_nextionSendInt("n0.val=",alarm.alarm.RTC_AlarmTime.RTC_Hours);
155  instruction_nextionSendInt("n1.val=",alarm.alarm.RTC_AlarmTime.RTC_Minutes);
156  instruction_nextionSendInt("bt0.val=",alarm.enable);
157  instruction_nextionSendStr("t11.txt=",rtc_dayToString(alarm.alarm.RTC_AlarmDateWeekDay));
158  instruction_nextionStop();
159  neopixel_setAllPixelRGB(0,0,255);
160  break;
161  default:
162  break;
163  }
164  }
