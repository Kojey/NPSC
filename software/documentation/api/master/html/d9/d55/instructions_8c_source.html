49  while(instruction_queue->size){
50  // fecth instruction
51  InstructionTypeDef _instruction = InstructionQueue_dequeue(instruction_queue);
52  // decode and execute instruction
53  switch(_instruction.instrution[0]){
54  /* External RTC instructions */
55  case 0x01:
56  // set clock
57  clock.date.RTC_Year=_instruction.instrution[1];
58  clock.date.RTC_Month=_instruction.instrution[2];
59  clock.date.RTC_Date=_instruction.instrution[3];
60  clock.date.RTC_WeekDay=_instruction.instrution[4];
61  clock.time.RTC_Hours=_instruction.instrution[5];
62  clock.time.RTC_Minutes=_instruction.instrution[6];
63  clock.time.RTC_Seconds=_instruction.instrution[7];
64  clock.time.RTC_H12=RTC_H12_AM;
65  rtc_setClockStruct(&clock);
66  break;
67  case 0x02:
68  // get clock
69  clock = rtc_getClockStruct();
70  if(b){
71  alarm_synchronize();
72  b=false;
73  }
74  instruction_nextionStart();
75  instruction_nextionSendInt("home.n0.val=",clock.time.RTC_Hours);

77  instruction_nextionSendStr("home.t1.txt=", rtc_dayToString(clock.date.RTC_WeekDay) );
78  instruction_nextionSendInt("home.n3.val=",clock.date.RTC_Date);
79  instruction_nextionSendStr("home.t2.txt=", rtc_monthToString(clock.date.RTC_Month) );
80  instruction_nextionSendInt("home.n4.val=",2000+clock.date.RTC_Year);
81  instruction_nextionStop();
82  break;
83  /* Alarm instructions */
84  case 0x10:
85  alarm = alarm_load(_instruction.instrution[1]);
86  // set alarm time params
87  //alarm.id=_instruction.instrution[1];
88  alarm.alarm.RTC_AlarmTime.RTC_Hours=_instruction.instrution[2];
89  alarm.alarm.RTC_AlarmTime.RTC_Minutes=_instruction.instrution[3];
90  alarm.alarm.RTC_AlarmDateWeekDaySel=rtc_weekdaySelTo32Bits(_instruction.instrution[4]);
91  alarm.alarm.RTC_AlarmDateWeekDay=_instruction.instrution[5];
92  alarm.repeat=_instruction.instrution[6];
93  alarm.alarm.RTC_AlarmMask=rtc_alarmMaskTo32Bits(_instruction.instrution[6]);
94  alarm.alarm.RTC_AlarmTime.RTC_H12=RTC_H12_AM;
95  alarm.alarm.RTC_AlarmTime.RTC_Seconds=0;
96  break;
97  case 0x11:
98  // set alarm extra params
99  alarm.ringtone=_instruction.instrution[2];
100  alarm.pattern=_instruction.instrution[3];
101  alarm.enable=_instruction.instrution[4];
102  alarm.fetched=_instruction.instrution[5];
103  break;
104  case 0x12:
105  // set alarm label
106  // get string from instruction
107  get_stringFromInstruction((char *)label_instruction, (char *)&_instruction.instrution[4],LABEL_INSTR_SIZE);
108  if (_instruction.instrution[2]==0)
109  strcpy(label,label_instruction);
110  else {
111  strcat(label,label_instruction);
112  strcpy(alarm.label,label);
113  // last instruction
114  if(_instruction.instrution[2]==2){
115  // save alarm
116  alarm_save(&alarm);
117  alarm_update(true);
118  }
119  }
120  break;
121  case 0x13:
122  // set alarm enable
123  alarm = alarm_load(_instruction.instrution[1]);
124  alarm.enable = _instruction.instrution[2];
125  alarm_save(&alarm);
126  break;
127  case 0x14:
128  // get alarm min params
129  instruction_nextionStart();

132  instruction_nextionSendInt("n0.val=",alarm.alarm.RTC_AlarmTime.RTC_Hours);
133  instruction_nextionSendInt("n1.val=",alarm.alarm.RTC_AlarmTime.RTC_Minutes);
134  instruction_nextionSendStr("t1.txt=",alarm.label);
135  instruction_nextionSendInt("bt0.val=",alarm.enable);

138  instruction_nextionSendInt("n2.val=",alarm.alarm.RTC_AlarmTime.RTC_Hours);
139  instruction_nextionSendInt("n3.val=",alarm.alarm.RTC_AlarmTime.RTC_Minutes);
140  instruction_nextionSendStr("t2.txt=",alarm.label);
141  instruction_nextionSendInt("bt1.val=",alarm.enable);
142  instruction_nextionStop();
143  break;
144  case 0x15:
145  // get alarm params
146  instruction_nextionStart();
147  alarm = alarm_load(_instruction.instrution[1]);
148  instruction_nextionSendStr("t6.txt=",alarm_repeatToString(alarm.repeat));

150  instruction_nextionSendStr("t8.txt=",alarm_patternToString(alarm.pattern));

155  instruction_nextionSendStr("t11.txt=",rtc_dayToString(alarm.alarm.RTC_AlarmDateWeekDay));
