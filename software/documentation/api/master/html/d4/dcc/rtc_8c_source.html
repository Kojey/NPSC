101 void rtc_transmissionStart(uint8_t direction){
102  // wait until I2C1 is not busy anymore
103  while(I2C_GetFlagStatus(RTC_I2CX, I2C_FLAG_BUSY));
104  // Send I2C1 START condition
105  I2C_GenerateSTART(RTC_I2CX, ENABLE);
106  // wait for I2C1 EV5 --> Slave has acknowledged start condition
107  while(!I2C_CheckEvent(RTC_I2CX, I2C_EVENT_MASTER_MODE_SELECT));
108  // Send slave Address for write
109  I2C_Send7bitAddress(RTC_I2CX, RTC_ADDRESS<<1, direction);//address, direction);
110  /* wait for I2C1 EV6, check if
111  * either Slave has acknowledged Master transmitter or
112  * Master receiver mode, depending on the transmission
113  * direction
114  */
115  if(direction == I2C_Direction_Transmitter){
116  while(!I2C_CheckEvent(RTC_I2CX, I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED));
117  }
118  else if(direction == I2C_Direction_Receiver){
119  while(!I2C_CheckEvent(RTC_I2CX, I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED));
120  }
121 }
