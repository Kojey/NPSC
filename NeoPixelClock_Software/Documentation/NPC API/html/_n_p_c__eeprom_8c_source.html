153 uint8_t eeprom_read(uint16_t address){
154  uint8_t value;
155  GPIO_ResetBits(GPIOB,GPIO_Pin_12); // CS Low -- start transmission
156  delay(1);
157 
158  // send read instruction
159  while(SPI_I2S_GetFlagStatus(SPI2, SPI_I2S_FLAG_TXE)==RESET);
160 
161  SPI_I2S_SendData(SPI2,READ); // read instruction
162  while(SPI_I2S_GetFlagStatus(SPI2,SPI_I2S_FLAG_RXNE)==RESET);
163  SPI_I2S_ReceiveData(SPI2); // junk data
164  // send 16 bits address register
165  while(SPI_I2S_GetFlagStatus(SPI2, SPI_I2S_FLAG_TXE)==RESET);
166  SPI_I2S_SendData(SPI2,address>>8); // send address of register
167  while(SPI_I2S_GetFlagStatus(SPI2,SPI_I2S_FLAG_RXNE)==RESET);
168  SPI_I2S_ReceiveData(SPI2); // junk data
169  SPI_I2S_SendData(SPI2,address);
170  while(SPI_I2S_GetFlagStatus(SPI2,SPI_I2S_FLAG_RXNE)==RESET);
171  SPI_I2S_ReceiveData(SPI2); // junk data
172 
173  // get data
174  SPI_I2S_SendData(SPI2,0x42);
175  while(SPI_I2S_GetFlagStatus(SPI2,SPI_I2S_FLAG_RXNE)==RESET);
176  value = SPI_I2S_ReceiveData(SPI2); // data
177 
178  GPIO_SetBits(GPIOB, GPIO_Pin_12); // CS High -- end transmission
179  delay(5000);
180  return value;
181 }
